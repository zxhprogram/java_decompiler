    public v15_TextBlock() {
    }
    public String html() {
        return "<html>
  <body>Hello</body>
</html>
";
    }
    public String withEscape() {
        String p0 = "line1 line2  
line3";
        return p0;
    }
    public String interpolate(String p0) {
        new Object[1][0] = p0;
        return "greeting: %s
".formatted(new Object[1]);
    }
}
