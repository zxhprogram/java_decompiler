public class v15_TextBlock {
    public v15_TextBlock() {
    }
    public String html() {
        return """
        <html>
          <body>Hello</body>
        </html>
        """;
    }
    public String withEscape() {
        String p1 = """
        line1 line2  
        line3
        """;
        return p1;
    }
    public String interpolate(String p0) {
        return """
        greeting: %s
        """.formatted(new Object[]{ p0 });
    }
}
